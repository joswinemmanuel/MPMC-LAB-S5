MOV SI, 3000H
MOV AL, [SI]
INC SI
MOV BL, [SI]
ADD AL, BL
INC SI
MOV [SI], AL
JC L1
INC SI
MOV [SI], AH
INT 3
L1:
INC SI
INC AH
MOV [SI], AH
INT 3
CODE ENDS
END START
